Future<void> setLogLevel(int logLevel) async {
  try {
    await _methodChannel
        .invokeMethod<void>('setLogLevel', {'level': logLevel});
  } on PlatformException catch (e) {
    print('Plugin error: ${e.message}');
  }
}
