Controls whether the Play button is enabled or not. Can not toggle this whilst the player is playing or paused.

void playbackEnabled({@required bool enabled})
{
        assert(__playState != _PlayState.playing && __playState != _PlayState.paused);
        setState(()
        {
                if (enabled == true)
                {
                        __playState = _PlayState.stopped;
                } else if (enabled == false)
                {
                        __playState = _PlayState.disabled;
                }
        });
}
